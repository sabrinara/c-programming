#include<stdio.h>
int main()
{
    int n,i;
    scanf("%d",&n);
    int a[n+1],even=0,odd=0;
    for(i=0;i<n;i++)
    {
        scanf("%d",& a[i]);
        if(a[i]%2==0)
        {
            even++;
        }
        else
        {
            odd++;
        }
    }

    printf("%d ",even);
    printf("%d\n",odd);
    return 0;
}

4.small and capital
#include<stdio.h>
int main()
{
    char str[1000];
    gets(str);
    int i,capital=0, small=0;
    for(i=0; str[i]!='\0'; i++)
    {
       if(str[i]>='A' && str[i]<='Z')
       {
           capital++;
       }
       else if(str[i]>='a' && str[i]<='z')
       {
           small++;
       }
    }
    printf("%d %d", capital,small);
    return 0;
}
